<COMMUNIY>

CREATE TABLE COMMUNITY (
	NO NUMBER PRIMARY KEY,
	TITLE VARCHAR2(60) NOT NULL,
	CONTENT VARCHAR2(3000) NOT NULL,
	WRITER VARCHAR2(15) NOT NULL,
	WDATE DATE DEFAULT SYSDATE,
	HIT NUMBER DEFAULT 0,
	LIKES NUMBER DEFAULT 0
)

CREATE SEQUENCE COM_SEQ START WITH 0 INCREMENT BY 1;

<DIARY>

CREATE TABLE DIARY (
	TITLE VARCHAR2(50) NOT NULL,
	WRITER VARCHAR2(15) NOT NULL,
	CONTENT VARCHAR2(3000) NOT NULL,
	WDATE DATE DEFAULT SYSDATE,
	ISPUBLIC NUMBER,
	LIKES NUMBER DEFAULT 0
	DIARY_CODE CHAR(6) PRIMARY KEY
)

CREATE SEQUENCE DIARY_SEQ START WITH 0 INCREMENT BY 1;

<FAMILY>

CREATE FAMILY_MEMBERS (
	NO NUMBER PRIMARY KEY,
	MEMBER_ID VARCHAR2(15),
	HOST_ID VARCHAR2(15) NOT NULL,
	FAMILY_NAME VARCHAR2(50) NOT NULL,
	PET_NAME VARCHAR2(30),
	CDATE DATE,
	STATUS NUMBER DEFAULT 0,
	LASTWALKWITH VARCHAR2(15),
	LASTWALKTIME DATE,
	LASTTREAT DATE,
	WHOGIVE VARCHAR2(15),
	MEMO VARCHAR2(300)
)

CREATE SEQUENCE FAM_SEQ START WITH 0 INCREMENT BY 1;

<LIKE>

CREATE TABLE LIKE_SET (
	POOL NUMBER,
	CODE CHAR(6),
	NAME VARCHAR2(15),
	WRITER VARCHAR2(15)
)

<MEMBER>

CREATE TABLE MEMBER (
	ID VARCHAR2(15) PRIMARY KEY,
	PWD VARCHAR2(20) NOT NULL,
	NICKNAME VARCHAR2(30) NOT NULL,
	EMAIL VARCHAR2(30) NOT NULL,
	PHONE VARCHAR2(20),
	PETNAME VARCHAR2(30),
	HASFAMILY NUMBER DEFAULT 0,
	FAMILYNAME VARCHAR2(30)
)

<MESSAGE>

CREATE TABLE MESSAGE (
	MSG_CODE CHAR(6) PRIMARY KEY,
	FROM_WHO VARCHAR2(15) NOT NULL,
	MSG_TIME DATE DEFAULT SYSDATE,
	MSG_CONTENT VARCHAR2(300),
	TO_WHO VARCHAR2(15) NOT NULL
)

CREATE SEQUENCE MSG_SEQ START WITH 1 INCREMENT BY 1;

<FILE>

CREATE TABLE FILE_COMMUNITY(
	CODE CHAR(6) NOT NULL,
	FILE_NAME VARCHAR2(100),
	FILE_ORG VARCHAR2(100),
	FILE_SYS VARCHAR2(100),
	INSERT_DATE DATE DEFAULT SYSDATE
)

CREATE TABLE FILE_DIARY(
	CODE CHAR(6) NOT NULL,
	FILE_NAME VARCHAR2(100),
	FILE_ORG VARCHAR2(100),
	FILE_SYS VARCHAR2(100),
	INSERT_DATE DATE DEFAULT SYSDATE,
	WRITER VARCHAR2(15),
